## Описание

Компонент группы радио.

<Story name='Песочница'>
    {React.createElement(() => {
        const [value, setValue] = React.useState({ one: false, two: false, three: false });
        const onChange = (_, payload) => {
            setValue({ ...value, [payload.name]: payload.checked });
        };
        const direction = select('direction', ['vertical', 'horizontal'], 'vertical');
        const error = text('error', '');
        const disabled = boolean('disabled', false);
        const label = text('label', 'Заголовок группы');
        return (
            <RadioGroup
                label={label}
                onChange={onChange}
                direction={direction}
                error={error}
                disabled={disabled}
                name='radioGroup'
            >
                <Radio label='Первый вариант' value='first' />
                <Radio label='Второй вариант' value='second' />
                <Radio label='Третий вариант' value='third' />
            </RadioGroup>
        );
    })}
</Story>
