        const [value, setValue] = React.useState({ one: false, two: false, three: false });
        const onChange = (_, payload) => {
            setValue({ ...value, [payload.name]: payload.checked });
        };
        const direction = select('direction', ['vertical', 'horizontal'], 'vertical');
        const error = text('error', '');
        const disabled = boolean('disabled', false);
        const label = text('label', '');
        const hint = text('hint', '');
        const size = select('Radio.size', ['s', 'm'], 's');
        return (
            <RadioGroup
                label={label}
                hint={hint}
                onChange={onChange}
                direction={direction}
                error={error}
                disabled={disabled}
                name='radioGroup'
            >
                <Radio size={size} label='Первый вариант' value='first' />
                <Radio size={size} label='Второй вариант' value='second' />
                <Radio size={size} label='Третий вариант' value='third' />
            </RadioGroup>
        );
    })}
