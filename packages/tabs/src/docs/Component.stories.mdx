        );

        } : {};
        return (
            <TabsComponent
                {...props}
                view={select('view', ['primary', 'secondary'], 'primary')}
                selectedId={selectedId}
                keepMounted={boolean('keepMounted')}
                onChange={handleChange}
                scrollable={boolean('scrollable', false)}
            >
                <Tab title='Таб 1' id='tab-1'>
