render(() => {
    const [sortDescBySum, setSortDescBySum] = React.useState(undefined);
    const [perPage, setPerPage] = React.useState(5);
    const [page, setPage] = React.useState(0);

    const handleSort = (...args) => {
        if (sortDescBySum) {
            setSortDescBySum(undefined);
        } else {
            setSortDescBySum(sortDescBySum === undefined ? false : true);
        }
    };

    const sortedData = React.useMemo(() => {
        if (sortDescBySum === undefined) return data;

        return [...data].sort((a, b) => (sortDescBySum ? b.sum - a.sum : a.sum - b.sum));
    }, [data, sortDescBySum]);

    const pagesCount = Math.ceil(sortedData.length / perPage);

    const currentPageData = React.useMemo(() => {
        return sortedData.slice(page * perPage).slice(0, perPage);
    }, [sortedData, page, perPage]);

    return (
        <Table.Wrapper style={{ margin: '24px 0 40px' }}>
            <Table>
                <Table.THead>
                    <Table.THeadCell width='122px' id='date' title='Дата'>
                        Дата
                    </Table.THeadCell>

                    <Table.THeadCell id='title' title='Контрагент'>
                        Контрагент
                    </Table.THeadCell>

                    <Table.THeadCell width='84px' id='number' title='Номер'>
                        Номер
                    </Table.THeadCell>

                    <Table.TSortableHeadCell
                        width='186px'
                        id='sum'
                        title='Сумма'
                        textAlign='right'
                        sorted={sortDescBySum !== undefined}
                        isSortedDesc={sortDescBySum}
                        onSort={handleSort}
                    >
                        Сумма
                    </Table.TSortableHeadCell>
                </Table.THead>
                <Table.TBody>
                    {currentPageData.map(row => (
                        <Table.TRow key={row.id}>
                            <Table.TCell>
                                <Typography.Text view='primary-small' tag='div'>
                                    {row.date}
                                </Typography.Text>
                            </Table.TCell>

                            <Table.TCell>
                                <Typography.Text view='primary-small' tag='div'>
                                    {row.title}
                                </Typography.Text>
                                <Typography.Text
                                    view='primary-small'
                                    color='secondary'
                                    tag='div'
                                    style={{ marginTop: 'var(--gap-2xs)' }}
                                >
                                    {row.subtitle}
                                </Typography.Text>
                            </Table.TCell>

                            <Table.TCell>
                                <Typography.Text view='primary-small'>{row.id}</Typography.Text>
                            </Table.TCell>

                            <Table.TCell>
                                <Amount
                                    value={row.sum}
                                    currency={row.currency}
                                    minority={100}
                                    view='withZeroMinorPart'
                                />
                            </Table.TCell>
                        </Table.TRow>
                    ))}
                </Table.TBody>
            </Table>

            <Table.Pagination
                perPage={perPage}
                currentPageIndex={page}
                possiblePerPage={[5, 10]}
                pagesCount={pagesCount}
                onPageChange={pageIndex => setPage(pageIndex)}
                onPerPageChange={value => {
                    setPage(0);
                    setPerPage(value);
                }}
            />
        </Table.Wrapper>
    );
});
