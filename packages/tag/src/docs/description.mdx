import Icon from '@alfalab/icons-glyph/CopyMIcon';

import { Tag } from '../';

### Тэг с иконкой

<Preview>
    {React.createElement(() => {
        const [checked, setChecked] = React.useState(false);
        const handleClick = () => setChecked(!checked);
        return (
            <Tag
                checked={checked}
                onClick={handleClick}
                rightAddons={<Icon />}
            >
                Нажми меня
            </Tag>
        );
    })}
</Preview>
