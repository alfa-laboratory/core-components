import { Container, Row, Col } from 'storybook/blocks';

Компонент тэг (чипсы).

```tsx live
render(() => {
    const SIZES = ['xxs', 'xs', 's', 'm', 'l', 'xl'];

    const [checked, setChecked] = React.useState(false);

    const handleClick = () => setChecked(!checked);

    return (
        <Tag
            size={select('size', SIZES, 'm')}
            disabled={boolean('disabled', false)}
            checked={checked || boolean('checked', false)}
            rightAddons={boolean('rightAddons', false) && <StarSIcon />}
            onClick={handleClick}
        >
            {text('label', 'Оплатить')}
        </Tag>
    );
});
```

### Тэг с иконкой

```tsx live
render(() => {
    const SIZES = ['xxs', 'xs', 's', 'm', 'l', 'xl'];

    return (
        <Container>
            <Row>
                {SIZES.map(size => (
                    <Col key={size}>
                        <Tag
                            checked={false}
                            rightAddons={size === 'xxs' ? <StarSIcon /> : <StarMIcon />}
                            size={size}
                        >
                            Tag
                        </Tag>
                    </Col>
                ))}
            </Row>

            <Row>
                {SIZES.map(size => (
                    <Col key={size}>
                        <Tag
                            checked={false}
                            leftAddons={size === 'xxs' ? <StarSIcon /> : <StarMIcon />}
                            size={size}
                        >
                            Tag
                        </Tag>
                    </Col>
                ))}
            </Row>
        </Container>
    );
});
```
