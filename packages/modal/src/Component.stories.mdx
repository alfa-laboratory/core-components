export const SimpleModal = () => {
    const [open, setOpen] = React.useState(false);
    const handleOpen = () => {
        setOpen(true);
    };
    const handleClose = () => {
        setOpen(false);
    };
    return (
        <div>
            <button type="button" onClick={handleOpen}>
                Open Modal
            </button>
            <ModalDesktop open={open} onClose={handleClose} footer={
                <div>
                    footer content
                </div>
            }>
                <div>
                    <p>Text in a modal</p>
                    <p>
                        Duis mollis, est non commodo luctus, nisi erat porttitor ligula.
                    </p>
                </div>
            </ModalDesktop>
        </div>
    );
};
