Компонент для построения графиков основан на библиотеке основаной на D3 с лицензией - MIT https://recharts.org.
В данный момент компонент поддерживает три типа графиков: `bar`, `line`, `area`.

## Данные для отрисовки

- `options.labels` - массив меток `string | number`
- `options.series` - массив объектов с параметрами грaфиков с обязательным полем `data`
- `options.series.data` - массив данных, который должен состоять из меток `label: string | number` и значений `value: number`

## Пример объекта с дефолтными значениями

```tsx
options: {
    /**
     * Oбязательное уникальное поле, такие елменты как `legend`
     * `linearGradient` используют его
     */
    id,

    /**
     * Компонент контейнера, позволяющий адаптировать диаграммы к размеру
     * родительского контейнера
     */
    responsiveContainer?: {
        /**
         * Значение для debounce функции при ресайзе
         */
        debounce?: 0,
    },

    /*
     * Диграмма для построения графиков с разными типами
     */
    composeChart: {
        /*
        * Отступы графика на момент инициализации компонента
        */
        initMargin?: {
            top?: 5,
            right?: 5,
            bottom?: 5,
            left?: 5,
        },

        /*
        * Максимальная ширина графиков типа bar
        */
        maxBarSize?,

        /*
        * Разрыв между двумя категориями столбцов,
        который может быть процентным или фиксированным значением
        */
        barCategoryGap?: '10%',

        /*
        * Разрыв между двумя столбцами одной категории
        */
        barGap?: 4,
    },

    /**
     * Отображение линий осей графика
     */
    cartesianGrid?: {
        /**
         * Отображение горизонтальной сетки
         */
        horizontal?: true,

        /**
         * Отображение вертикальной сетки
         */
        vertical?: true,

        /**
         * Штрихи для осей
         */
        strokeDashArray?,

        /**
         * Цвет разметки сетки
         */
        stroke?,

        /**
         * Цвет заливки графика
         */
        fill?,
    },

    /**
     * Настройки оси Х
     */
    xAxis: {
        /**
         * Скрыть ось
         */
        hide?: false,

        /**
         * Отображение линии оси.
         */
        axisLine?: true,

        /**
         * Тип оси.
         */
        type: 'category',

        /**
         * Разрешить значениям оси быть десятичными или нет.

        allowDecimals?: true,

        /**
         * Разрешить или нет на оси дублировать категории,
         * если "type" оси - "category".
         */
        allowDuplicatedCategory?: true,

        /**
         * Угол наклона значений оси.
         */
        angle?: 0,

        /**
         * Количество делений оси. Не используется,
         * если "тип" - "категория".
         */
        tickCount?: 5,

        /**
         * Интревал линий деления оси
         */
        interval?: 'preserveEnd',

        /**
         * Минимальный зазор между двумя соседними делениями оси.
         */
        minTickGap?: 5,

        /**
         * Отображение линий деления оси
         */
        tickLine?: true,

        /**
         * Размер линий деления оси
         */
        tickSize?: 6,

        /**
         * Если установлено значение true,
         * галочки переворачиваются вокруг оси,
         * отображая метки внутри диаграммы, а не снаружи.
         */
        mirror?: false,

        /**
         * Отображение горизонтальной сетки
         */
        reversed?: false,

        /**
         * Реверсировать ось или нет
         */
        tickMargin?: 0,

        /**
         * Функция форматирования значения деления оси.
         */
        tickFormatter?,

        /**
         * Ключ данных, отображаемых на оси.
         */
        dataKey,

        /**
         * Ориентация оси
         */
        orientation?: 'bottom',

        /**
         * Тип деления оси
         */
        tickType?,
    },

    /**
     * Настройки оси Y
     */
    yAxis: {
        /**
         * Скрыть ось
         */
        hide?: false,

        /**
         * Отображение линии оси.
         */
        axisLine?: true,

        /**
         * Тип оси.
         */
        type: 'number',

        /**
         * Разрешить значениям оси быть десятичными или нет.
         */
        allowDecimals?: true,

        /**
         * Разрешить или нет на оси дублировать категории,
         * если "type" оси - "category".
         */
        allowDuplicatedCategory?: true,

        /**
         * Угол наклона значений оси.
         */
        angle?: 0,

        /**
         * Количество делений оси.
         * Не используется, если "тип" - "категория".
         */
        tickCount?: 5,

        /**
         * Интревал линий деления оси
         */
        interval?: 'preserveEnd',

        /**
         * Минимальный зазор между двумя соседними делениями оси.
         */
        minTickGap?: 5,

        /**
         * Отображение линий деления оси
         */
        tickLine?: true,

        /**
         * Размер линий деления оси
         */
        tickSize?: 6,

        /**
         * Если установлено значение true,
         * галочки переворачиваются вокруг оси,
         * отображая метки внутри диаграммы, а не снаружи.
         */
        mirror?: false,

        /**
         * Отображение горизонтальной сетки
         */
        reversed?: false,

        /**
         * Реверсировать ось или нет
         */
        tickMargin?: 0,

        /**
         * Функция форматирования значения деления оси.
         */
        tickFormatter?,

        /**
         * Ключ данных, отображаемых на оси.
         */
        orientation?: 'bottom',
    },

    /**
     * Всплывающаяся подсказка
     */
    tooltip?: {
        /**
         * Разделитель между именем и значением
         */
        separator?,

        /**
         * Размер смещения между положением всплывающей
         * подсказки и активной позицией
         */
        offset?: 10,

        /**
         * Когда элемент полезной нагрузки имеет значение null или undefined,
         * этот элемент не отображается
         */
        filterNull: true,

        /**
         * Если установлено false, курсор не будет отображаться, когда всплывающая подсказка активна.
         * Если установлен объект, опция - это конфигурация курсора.
         * Если установлен элемент React, опция является настраиваемым элементом реакции курсора рисования.
         */
        cursor?: {
            /**
             * Цвет линии
             */
            stroke: cursorStroke,
    
            /**
             * Ширина линии
             */
            strokeWidth: cursorStrokeWidth,
        },

        /**
         * Если это поле установлено, положение всплывающей подсказки
         * будет фиксированным и больше не будет перемещаться.
         */
        position?: {
            x,
            y,
        };

        /**
         * Если установлено false, анимация всплывающей подсказки будет отключена.
         */
        isAnimationActive?: true,

        /**
         * Указывает, когда должна начинаться анимация,
         * единица измерения этого параметра - мс.
         */
        animationBegin?: 0,

        /**
         * Определяет продолжительность анимации,
         * единица измерения этого параметра - мс.
         */
        animationDuration?: 1500,

        /**
         * Тип функции плавности
         */
        animationEasing?: 'ease',

        /**
         * Отображение стрелки тултипа
         */
        arrow?: false,
    };

    /**
     * Компонент маштабирования графика
     */
    brush?: {
        /**
         * Ключ данных
         */
        dataKey,

        /**
         * Ширина ползунков
         */
        travellerWidth?: 5,

        /**
         * Начальный индекс интервала показа
         */
        startIndex?,

        /**
         * Конечный индекс интервала показа
         */
        endIndex?,

        /**
         * Отступ
         */
        brushMargin?: 0,

        /**
         * Высота
         */
        height?: 40,

        /**
         * Форматирование значений
         */
        tickFormatter?,

        /**
         * Всегда показывать значение текущего интервала
         */
        alwaysShowText?: false,

        /**
         * Цвет
         */
        stroke?: brushStroke,
    }

    /**
     * Компонент подписи графиков
     */
    legend?: {
        /**
         * Выравнивание элементов легенды в вертикальном направлении
         */
        verticalAlign?: 'middle',

        /**
         * Функция форматирования каждого текста в легенде
         */
        formatter?,

        /**
         * Выравнивание элементов легенды в горизонтальном направлении
         */
        align?: 'center',

        /**
         * Вертикальный отступ
         */
        marginTop?: 0,

        /**
         * Высота иконок
         */
        iconHeight?: 16,
    },

    /**
     * Mассив меток
     */
    labels: Array<string | number>,

    /**
     * Mассив объектов с параметрами грaфиков с обязательным полем 
     */
    series: [
        /*
         * Пример графика типа bar
         */
        {
            /*
             * Скрыть график
             */
            hide?: false,

            /*
             * Скрыть график в легенде
             */
            hideLegend?: false,

            /*
             * Скрыть график в тултипe
             */
            hideTooltip?: false,

            /*
             * Z-index графика
             */
            zIndex,

            /*
             * Тип графика
             */
            chart: 'bar',

            /**
             * Радиус закругления углов графика типа bar
             */
            radius?: number || {
                topLeft?,
                topRight?,
                bottomLeft?,
                bottomRight?,
            },

            /*
             * Тип иконки для графика
             */
            icon,

            /*
             * Параметры графика
             */
            properties: {

                /*
                * Название графика (отображается в легендах и тултипе)
                */
                name,

                /*
                * Ключ данных
                */
                dataKey,

                /*
                * Форматирование значения графика для тултипа
                */
                formatter?,

                /*
                * Цвет заливки графика
                */
                fill,

                /*
                * Цвет бордера графика
                */
                strokeWidth?,
            },

            /**
             * Данные для построения графика
             */
            data: [
                /**
                 * Метка
                 */
                label,

                /**
                 * Значение
                 */
                value
            ];
        },

        /*
         * Пример графика типа line
         */
        {
            /*
             * Скрыть график
             */
            hide?: false,

            /*
             * Скрыть график в легенде
             */
            hideLegend?: false,

            /*
             * Скрыть график в тултипe
             */
            hideTooltip?: false,

            /*
             * Z-index графика
             */
            zIndex,

            /*
             * Тип графика
             */
            chart: 'line',

            /*
             * Тип иконки для графика
             */
            icon,

            /*
             * Параметры графика
             */
            properties: {

                /*
                 * Название графика (отображается в легендах и тултипе)
                 */
                name,

                /*
                 * Ключ данных
                 */
                dataKey,

                /*
                 * Цвет линии графика
                 */
                stroke?: '#3182bd',

                /*
                 * Отображение точек на графике
                 */
                dot?,

                /*
                 * Параметры для точек грфика (если dot: true)
                 */
                dotSettings: [{

                    /*
                     * Значение media query
                     */
                    media,

                    /*
                     * Значение маштаба при ховере
                     */
                    scale,

                    /*
                     * Начальный значения маштаба
                     */
                    initScale,

                    /*
                     * Ширина точки
                     */
                    width,

                    /*
                     * Высота точки
                     */
                    height,
                }],

                /*
                 * Наследование цвета
                 */
                inheritStroke?: false,

                /*
                 * Форматирование значения графика для тултипа
                 */
                formatter?,

                /*
                 * Тип линии
                 */
                type? 'linear',

                /*
                 * Толщина линии
                 */
                strokeWidth?: 1,

                /*
                 * Прерывистость линии
                 */
                strokeDasharray?,
            },

            /*
             *  Массив данных для отрисовки
             */
            data: [

                /**
                 * Метка
                 */
                label,

                /**
                 * Значение
                 */
                value,
            ],
        },

        /*
         * Пример графика типа area
         */
        {
            /*
             * Скрыть график
             */
            hide?: false,

            /*
             * Скрыть график в легенде
             */
            hideLegend?: false,

            /*
             * Скрыть график в тултипe
             */
            hideTooltip?: false,

            /*
             * Z-index графика
             */
            zIndex,

            /*
             * Тип графика
             */
            chart: 'ara',

            /*
             * Тип иконки для графика
             */
            icon,

            /*
             * Оффсет для градиента (только для типа area)
             */
            offset,

            /**
             * Параметры градиента
             */
            gradient: {

                /**
                 * Индефикатор градиента
                 */
                gid,

                /**
                 * Параметра точек для градиента
                 */
                points: [

                    /**
                     * Отступ
                     */
                    offset: number,

                    /**
                     * Цвет
                     */
                    stopColor: string,

                    /**
                     * Прозрачность
                     */
                    stopOpacity: number,

            };

            /*
             * Параметры графика
             */
            properties: {

                /*

                 */
                name,

                /*
                 * Ключ данных
                 */
                dataKey,

                /*
                 * Цвет линии графика
                 */
                stroke?: '#3182bd',

                /*
                 * Отображение точек на графике
                 */
                dot?,

                /*
                 * Параметры для точек грфика (если dot: true)

                dotSettings: [{

                    /*
                     * Значение media query
                     */
                    media,

                    /*

                    /*

                     */
                    initScale,

                    /*
                     * Ширина точки

                /*

                 */
                inheritStroke?: false,

                 */
                formatter?,

                /*
                 * Тип линии

                /*
                 * Толщина линии
                 */
                strokeWidth?: 1,

                /*

                 */

            },

            /**
             * Данные для построения графика
             */

                 * Метка
                 */
                label,
```
