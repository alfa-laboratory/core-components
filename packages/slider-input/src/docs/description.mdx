const min = 1;
const max = 5;
const step = 1;
const steps = ['1 год', '2 года', '3 года', '4 года', '5 лет'];

const normalizeValue = (value, min, max) =>
    value === '' ? '' : Math.min(max, value.toString().replace(/[^\d-]/g, ''));

const getSuffix = years => {
    if (years <= 0) return '';
    if (years === 1) return ' год';
    if (years > 1 && years < 5) return ' года';
    if (years >= 5) return ' лет';
};

render(() => {
    const [value, setValue] = React.useState('');

    const handleChange = (event, { value }) => setValue(normalizeValue(value, min, max));

    return (
        <SliderInput
            value={value}
            min={min}
            max={max}
            step={step}
            onChange={handleChange}
            block={true}
            label='Срок кредита'
            steps={steps}
            Input={SuffixInput}
            customInputProps={{
                suffix: getSuffix(value),
            }}
        />
    );
});

## AmountSliderInput

```tsx live
// import { AmountInput } from '@alfalab/core-components/amount-input';

const min = 100000;
const max = 600000;
const step = 1000;
const steps = ['100 тыс.', '600 тыс.'];

render(() => {
    const [value, setValue] = React.useState('');

    const handleInputChange = (event, { value }) => {
        setValue(value !== '' ? (+value / 100).toString() : '');
    };

    const handleSliderChange = (event, { value }) => {
        setValue(value);
    };

    const handleBlur = () => {
        if (value !== '') {
            setValue(Math.max(min, Math.min(max, +value)));
        }
    };

    return (
        <SliderInput
            value={value !== '' ? value * 100 : ''}
            sliderValue={value}
            onInputChange={handleInputChange}
            onSliderChange={handleSliderChange}
            onBlur={handleBlur}
            min={min}
            max={max}
            step={step}
            Input={AmountInput}
            steps={steps}
            label='Сумма кредита'
        />
    );
});
```
