Компонент drag-and-drop контейнер для прикрепления файлов.

```tsx live
render(() => {
    const [filesList, setFilesList] = React.useState([]);

    const handleDrop = files => {
        setFilesList(
            Array.from(files)
                .map(file => file.name)
                .filter(Boolean),
        );
    };

    return (
        <Dropzone
            error={boolean('error', false)}
            overlayVisible={boolean('overlayVisible', false)}
            onDrop={handleDrop}
        >
            <div
                style={{
                    width: 400,
                    height: 200,
                    display: 'flex',
                    alignItems: 'center',
                    justifyContent: 'center',
                }}
            >
                {filesList.length > 0 ? (
                    <ul>
                        {filesList.map(file => (
                            <li key={file}>{file}</li>
                        ))}
                    </ul>
                ) : (
                    <span>children</span>
                )}
            </div>
        </Dropzone>
    );
});
```
